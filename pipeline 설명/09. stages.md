![image](https://user-images.githubusercontent.com/24771449/67619562-2094f200-f838-11e9-85ab-bf69c83bae3a.png)

- Javascript 작성
```javascript
/* 피자 최종 가격 계산 하기 */
총계산금액 = 피자가격 * 피자수량;

/* 천단위 쉼표 처리 */
피자가격 = 피자가격.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
총계산금액 = 총계산금액.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
```
